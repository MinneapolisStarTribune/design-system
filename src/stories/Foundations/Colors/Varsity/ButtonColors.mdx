import { Meta, ColorPalette, ColorItem } from '@storybook/addon-docs/blocks';
import { varsityColorCategories } from '../brandColorsData';

<Meta title="Foundations/Colors/Varsity/Button Colors" />

{varsityColorCategories
  .filter((category) => category.name === 'Button')
  .map((category) => (
    <div key={category.name}>
      <h1>{category.name}</h1>
      {category.description && <p>{category.description}</p>}
      <h2>Overview</h2>
      {category.overview && category.overview.length > 0 && (
        <ul>
          {category.overview.map((bullet, index) => (
            <li key={index}>{bullet}</li>
          ))}
        </ul>
      )}
      <ColorPalette>
        {category.colors.map((color) => (
          <ColorItem
            key={color.name}
            title={`${category.name}: ${color.name}`}
            subtitle={JSON.stringify(color.value)}
            colors={{ [color.tokenName]: color.value }}
          />
        ))}
      </ColorPalette>
    </div>
  ))}
