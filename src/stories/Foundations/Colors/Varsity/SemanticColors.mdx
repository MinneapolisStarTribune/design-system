import { Meta, ColorPalette, ColorItem } from '@storybook/blocks';
import { varsityColorCategories } from '../brandColorsData';

<Meta title="Foundations/Colors/Varsity/Semantic Colors" />

{varsityColorCategories
  .filter((category) => category.name === 'Semantic')
  .map((category) => (
    <div key={category.name}>
      <h1>{category.name}</h1>
      {category.description && <p>{category.description}</p>}
      <ColorPalette>
        {category.colors.map((color) => (
          <ColorItem
            key={color.name}
            title={`${category.name}: ${color.name}`}
            subtitle={color.description}
            colors={{ [color.tokenName]: color.value }}
          />
        ))}
      </ColorPalette>
    </div>
  ))}
